Arrays & Strings

1. Check if String is Palindrome

Easy

Write a program to check if a string is a palindrome (ignoring non-alphanumeric characters and case).

// JavaScript Solution
function isPalindrome(str) {
    // Remove non-alphanumeric and convert to lowercase
    const cleaned = str.replace(/[^a-z0-9]/gi, '').toLowerCase();
    
    // Compare with reversed string
    const reversed = cleaned.split('').reverse().join('');
    return cleaned === reversed;
}

// Alternative two-pointer approach
function isPalindromeTwoPointer(str) {
    const cleaned = str.replace(/[^a-z0-9]/gi, '').toLowerCase();
    let left = 0;
    let right = cleaned.length - 1;
    
    while (left < right) {
        if (cleaned[left] !== cleaned[right]) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}

// Test cases
console.log(isPalindrome("A man, a plan, a canal: Panama")); // true
console.log(isPalindrome("race a car")); // false
Time Complexity: O(n) | Space Complexity: O(n)

2. String Compression

Medium

Given a string like "abbcddeee", output "ab2cd2e3" (compress repeated adjacent characters with counts).

// JavaScript Solution
function compressString(str) {
    if (!str) return '';
    
    let result = '';
    let count = 1;
    
    for (let i = 0; i < str.length; i++) {
        // Check if current char matches next char
        if (i + 1 < str.length && str[i] === str[i + 1]) {
            count++;
        } else {
            // Add character and count (if > 1) to result
            result += str[i];
            if (count > 1) {
                result += count;
            }
            count = 1;
        }
    }
    
    return result;
}

// Test cases
console.log(compressString("abbcddeee")); // "ab2cd2e3"
console.log(compressString("aabcccccaaa")); // "a2bc5a3"
console.log(compressString("abc")); // "abc"
Time Complexity: O(n) | Space Complexity: O(n)

3. Find Largest Element in Array

Easy

Find the largest element in an unsorted array.

// JavaScript Solution
function findLargest(arr) {
    if (arr.length === 0) return null;
    
    let max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

// Using built-in method
function findLargestBuiltIn(arr) {
    return Math.max(...arr);
}

// Test cases
console.log(findLargest([3, 1, 4, 1, 5, 9, 2, 6])); // 9
console.log(findLargest([-5, -2, -10, -1])); // -1
Time Complexity: O(n) | Space Complexity: O(1)

4. Reverse a String

Medium

Reverse a string in-place (if possible in your language).

// JavaScript Solution - Multiple Approaches
function reverseString1(str) {
    return str.split('').reverse().join('');
}

function reverseString2(str) {
    let reversed = '';
    for (let i = str.length - 1; i >= 0; i--) {
        reversed += str[i];
    }
    return reversed;
}

function reverseString3(str) {
    return [...str].reduceRight((acc, char) => acc + char, '');
}

// Test cases
console.log(reverseString1("hello")); // "olleh"
console.log(reverseString2("world")); // "dlrow"
Time Complexity: O(n) | Space Complexity: O(n)

Data Structures

5. Implement a Stack

Medium

Implement a stack with push, pop, peek, and isEmpty operations.

// JavaScript Solution
class Stack {
    constructor() {
        this.items = [];
    }
    
    push(element) {
        this.items.push(element);
    }
    
    pop() {
        if (this.isEmpty()) {
            return null;
        }
        return this.items.pop();
    }
    
    peek() {
        if (this.isEmpty()) {
            return null;
        }
        return this.items[this.items.length - 1];
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
    
    size() {
        return this.items.length;
    }
    
    clear() {
        this.items = [];
    }
}

// Test cases
const stack = new Stack();
stack.push(10);
stack.push(20);
stack.push(30);
console.log(stack.peek()); // 30
console.log(stack.pop());  // 30
console.log(stack.size()); // 2
Time Complexity: O(1) for all operations | Space Complexity: O(n)

6. Implement a Queue

Medium

Implement a queue with enqueue, dequeue, front, and isEmpty operations.

// JavaScript Solution
class Queue {
    constructor() {
        this.items = [];
    }
    
    enqueue(element) {
        this.items.push(element);
    }
    
    dequeue() {
        if (this.isEmpty()) {
            return null;
        }
        return this.items.shift();
    }
    
    front() {
        if (this.isEmpty()) {
            return null;
        }
        return this.items[0];
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
    
    size() {
        return this.items.length;
    }
}

// Test cases
const queue = new Queue();
queue.enqueue(1);
queue.enqueue(2);
queue.enqueue(3);
console.log(queue.front());   // 1
console.log(queue.dequeue()); // 1
console.log(queue.size());    // 2
Time Complexity: O(1) for enqueue, O(n) for dequeue | Space Complexity: O(n)

7. Detect Cycle in Linked List

Hard

Given a linked list, determine if it has a cycle in it.

// JavaScript Solution - Floyd's Cycle Detection
class ListNode {
    constructor(val) {
        this.val = val;
        this.next = null;
    }
}

function hasCycle(head) {
    if (!head || !head.next) {
        return false;
    }
    
    let slow = head;
    let fast = head;
    
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        
        if (slow === fast) {
            return true;
        }
    }
    
    return false;
}

// Test case
const node1 = new ListNode(3);
const node2 = new ListNode(2);
const node3 = new ListNode(0);
const node4 = new ListNode(-4);

node1.next = node2;
node2.next = node3;
node3.next = node4;
node4.next = node2; // Creates cycle

console.log(hasCycle(node1)); // true
Time Complexity: O(n) | Space Complexity: O(1)

Searching & Sorting

8. Binary Search

Medium

Implement binary search to find an element in a sorted array.

// JavaScript Solution - Iterative
function binarySearch(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        
        if (arr[mid] === target) {
            return mid;
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return -1; // Not found
}

// Recursive approach
function binarySearchRecursive(arr, target, left = 0, right = arr.length - 1) {
    if (left > right) {
        return -1;
    }
    
    const mid = Math.floor((left + right) / 2);
    
    if (arr[mid] === target) {
        return mid;
    } else if (arr[mid] < target) {
        return binarySearchRecursive(arr, target, mid + 1, right);
    } else {
        return binarySearchRecursive(arr, target, left, mid - 1);
    }
}

// Test cases
const arr = [1, 3, 5, 7, 9, 11, 13, 15];
console.log(binarySearch(arr, 7));  // 3
console.log(binarySearch(arr, 10)); // -1
Time Complexity: O(log n) | Space Complexity: O(1) iterative, O(log n) recursive

Technical Concepts (Common Interview Questions)

Shallow vs Deep Copy (C++)

Shallow Copy: Copies the object's pointer, both objects share the same memory address. Changes in one affect the other.

Deep Copy: Creates a new copy of the object with separate memory. Changes in one don't affect the other.

// C++ Example
class Example {
public:
    int* data;
    
    // Shallow copy (default copy constructor)
    Example(const Example& obj) {
        data = obj.data; // Both point to same memory
    }
    
    // Deep copy
    Example deepCopy(const Example& obj) {
        Example newObj;
        newObj.data = new int;
        *newObj.data = *obj.data; // Separate memory
        return newObj;
    }
};

Virtual Keyword in C++

The virtual keyword enables runtime polymorphism. It allows a derived class to override a base class method.

// C++ Example
class Base {
public:
    virtual void display() {
        cout << "Base class" << endl;
    }
};

class Derived : public Base {
public:
    void display() override {
        cout << "Derived class" << endl;
    }
};

// Usage
Base* ptr = new Derived();
ptr->display(); // Outputs: "Derived class"

V-Table (Virtual Table): A lookup table of function pointers used to resolve virtual function calls at runtime.

OOP Principles

  • Encapsulation: Bundling data and methods that operate on that data within a class
  • Inheritance: Creating new classes from existing ones (is-a relationship)
  • Polymorphism: Ability to take multiple forms (method overloading/overriding)
  • Abstraction: Hiding complex implementation details and showing only essential features
